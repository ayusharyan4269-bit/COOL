#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;

        string s;
        cin >> s;

        int ans = 0;

        for (int i = 0; i < n;) {

            if (s[i] == 'W') {
                i++;
            }
            else {
                ans++;
                i += k;
            }
        }

        cout << ans << endl;
    }

    return 0;
}
